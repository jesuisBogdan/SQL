 SELECT title, score
 FROM hacker_news
 ORDER BY score DESC
 LIMIT 5;

 SELECT SUM(score)
 FROM hacker_news;

 SELECT user, SUM(score)
 FROM hacker_news
 GROUP BY user
 HAVING SUM(score) > 200
 ORDER BY 2 DESC;

 SELECT user, COUNT(*)
 FROM hacker_news
 WHERE url LIKE '%watch?v=dQw4w9WgXcQ%'
 GROUP BY 1
 ORDER BY 2 DESC;


 SELECT CASE
    WHEN url LIKE '%github%' then 'GitHub'
    WHEN url LIKE '%medium%' then 'Medium'
    WHEN url LIKE '%nytimes%' then 'New York Times'
    ELSE 'Other'
    END AS 'Source',
    COUNT(*)
  FROM hacker_news
  GROUP BY 1;

  SELECT timestamp, strftime('%H', timestamp)
  FROM hacker_news
  GROUP BY 1
  LIMIT 10;

  SELECT strftime('%H', timestamp) AS 'HOUR', ROUND(AVG(score), 2) AS 'Average Score', COUNT(*) AS 'Number of Stories'
  FROM hacker_news
  WHERE timestamp IS NOT NULL
  GROUP BY 1
ORDER BY 2;
